# 日志处理

```go
func Logout(v ...interface{}) {
    t := time.Now()
	logFileName := strings.Join([]string{"log/", t.Format("2006-01-02-15-04-05"), ".log"}, "")
	file, err := os.OpenFile(logFileName, os.O_APPEND|os.O_CREATE, 666)
	if err != nil {
		log.Fatalln(err)
	}
	defer file.Close()
	logger := log.New(file, "", log.LstdFlags|log.Lshortfile) // 日志文件格式:log包含时间及文件行数
	logger.Println(v)
}
```
